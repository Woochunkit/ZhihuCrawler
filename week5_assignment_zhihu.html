import re
def find_chinese(file):
    pattern = re.compile(r'[^\u4e00-\u9fa5]')
    chinese = re.sub(pattern, '', file)

In [74]:
results1=[]
def get_words_num(temp):
    seg_list = jieba.cut(find_chinese(temp))
    seg_string = ", ".join(seg_list)
    results = seg_string.split(',')
    for i in range(len(results)):
        if len(results[i]) > 2:
            results1.append(results[i])
    results_dict = dict(collections.Counter(results1))
    #按照value进行排序
    results_dict = {k: v for k, v in sorted(results_dict.items(), key=lambda item: item[1], reverse=True)}
    #筛选前十个
    return({item[0]:item[1] for i,item in enumerate(results_dict.items()) if i < 100})
